题目：

[Daily Temperatures](https://leetcode.com/problems/daily-temperatures/)

解答：

```golang
func dailyTemperatures(T []int) []int {
	length := len(T)
	list := make([]int, length)
	for i := 0; i < length-1; i ++ {
		find := false
		j := 0
		for j = i + 1; j < length; j ++ {
			if T[j] <= T[i] {
				continue
			} else {
				find = true
				break
			}
		}
		if find {
			list[i] = j - i
		} else {
			list[i] = 0
		}
	}
	// 最后一位始终为0
	list[length-1] = 0
	return list
}
```
